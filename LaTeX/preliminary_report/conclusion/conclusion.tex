            \times & \times &        &        \\

            \times &        &        &        \\
            \times & \Ddots &        &        \\

        \end{bNiceMatrix}

            \times & \times &        &        \\
                   & \Ddots & \Ddots &        \\

                   &        &        & \times
        \end{bNiceMatrix}
        = LU \ \ (\text{or}\ LL^{\T} \ \text{if } \Delta  \ \text{is posi. def.})
    \end{align*}
